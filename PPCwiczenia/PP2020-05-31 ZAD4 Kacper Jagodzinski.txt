# y = ax^2 + bx + c
import math

a = float(input("Podaj wartoœæ a: "))
b = float(input("Podaj wartoœæ b: "))
c = float(input("Podaj wartoœæ c: "))

#obliczanie pierwiastkow

delta = ((b*b) - (4*a*c))

if delta == 0:
  x0 = ((-b)/(2*a))
  print("Delta = 0 => x0 = " + str(x0))
elif delta > 0:
  x1 = (-b-math.sqrt(delta))/(2*a)
  x2 = (-b+math.sqrt(delta))/(2*a)  
  print("Delta > 0 => x1 = % 0.1f x2 = % 0.1f" %(x1, x2))
else:
  print("Delta<0 brak rozwiazan rzeczywistych")
